# CinaGroup AI News Briefing Auto-Publish Schedule
# Publishes 30 minutes before each briefing time (Asia/Shanghai)

# Morning briefing (06:00) → publish at 05:30
30 5 * * * /root/.openclaw/workspace/scripts/news-briefing-auto-publish.sh 06 >> /root/.openclaw/workspace/logs/news-briefing-06.log 2>&1

# Evening briefing (18:00) → publish at 17:30
30 17 * * * /root/.openclaw/workspace/scripts/news-briefing-auto-publish.sh 18 >> /root/.openclaw/workspace/logs/news-briefing-18.log 2>&1

# Health check - runs every hour to verify briefings are published
